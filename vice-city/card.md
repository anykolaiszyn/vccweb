/* QR Card Landing — lightweight analytics event tracking */
(function () {
  'use strict';

  /* ---- UTM / source detection ---- */
  function tagSession() {
    var params = new URLSearchParams(window.location.search);
    var source = params.get('utm_source') || 'business-card-qr';
    sessionStorage.setItem('vcc_source', source);
  }

  /* ---- Simple click tracker ---- */
  function trackClick(label) {
    /* Replace with your analytics provider as needed.
       Currently logs to console in dev; wire to GA4, Plausible, etc. */
    if (typeof gtag === 'function') {
      gtag('event', 'card_cta_click', {
        event_category: 'QR Landing',
        event_label: label,
        non_interaction: false
      });
    }
    if (typeof window._paq !== 'undefined') {
      window._paq.push(['trackEvent', 'QR Landing', 'Click', label]);
    }
  }

  /* ---- Form submission tracker ---- */
  function trackFormSubmit() {
    if (typeof gtag === 'function') {
      gtag('event', 'card_inquiry_submit', {
        event_category: 'QR Landing',
        event_label: 'inquiry-form',
        non_interaction: false
      });
    }
  }

  /* ---- Init ---- */
  document.addEventListener('DOMContentLoaded', function () {
    tagSession();

    /* Track all data-track elements */
    document.querySelectorAll('[data-track]').forEach(function (el) {
      el.addEventListener('click', function () {
        trackClick(el.getAttribute('data-track'));
      });
    });

    /* Track form submission */
    var form = document.getElementById('card-inquiry-form');
    if (form) {
      form.addEventListener('submit', function () {
        trackFormSubmit();
      });
    }

    /* Mark as QR scan visit (fires once per session) */
    if (!sessionStorage.getItem('vcc_qr_tracked')) {
      sessionStorage.setItem('vcc_qr_tracked', '1');
      if (typeof gtag === 'function') {
        gtag('event', 'qr_scan_visit', {
          event_category: 'QR Landing',
          event_label: 'business-card',
          non_interaction: true
        });
      }
    }
  });
}());
</script>
